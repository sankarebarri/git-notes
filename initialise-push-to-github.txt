# Create a new repository on GH(preferably as the same name as the project name on your local computer)
# Navigate to the root of your local repository

# Then on git on the project path:
    -> git init
    -> git add .
    -> git commit -m 'message'
    -> git remote add origin <PASTE_YOUR_REPO_URL_HERE>
    -> git push -u origin main


# unstaging(i.e. doing git add . but not yet git commit -m 'msg')
    -> git reset

# unstaging(un adding and un-commiting)
    # if it's the first and only commit
    -> git update-ref -d HEAD
    -> git reset

    # if there's been a previous commit before this
    -> git reset HEAD~1
